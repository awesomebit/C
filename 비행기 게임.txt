#if 01
#include <stdio.h>

int T;
char map[13 + 10][5 + 10];
int M, N;

int di[] = { -1, -1, -1 };
int dj[] = { 0, -1, 1 };

int score;

void DFS(int n, int sum, int i, int j, int attack)
{
	if (n >= 13)
	{	
		if (score < sum)
		{
			score = sum;
		}
		return;
	}

	int ni, nj;

	int k, l, m;
	for (k = 0; k < 3; k++)
	{
		ni = i + di[k];
		nj = j + dj[k];

		if (ni < 1 || nj < 1 || nj > 5) continue;

		int s = 0;
		int b = 0;
		if (map[ni][nj] == '*') s++;
		else if (map[ni][nj] == 'X') b++;

		DFS(n + 1, sum + s * 10 - b * 7, ni, nj, 1);

		if (attack == 1)
		{
			for (l = ni - 4; l <= ni; l++)
			{
				if (l < 1) continue;
				if (map[l][nj] == '*') continue;
				map[l][nj] = '0';
			}
			DFS(n + 1, sum + s * 10, ni, nj, 0);
		}
	}
}

void main(void)
{
	scanf("%d", &T);
	int i, j;
	int k;
	for (k = 1; k <= T; k++)
	{
		for (i = 1; i <= 13; i++)
		{
			for (j = 1; j <= 5; j++)
			{
				scanf(" %c", &map[i][j]);
			}
		}

		score = 0;
		DFS(1, 0, 13, 3, 1);
		printf("%d", score);
	}
}
#endif
