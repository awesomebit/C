#include <stdio.h>

int N;
int a[12 + 10][12 + 10];
int chk[12 + 10];
int sol = 0x7fffffff;

int DFS(int n, int sum, int city)
{
	int i;

	//가지치기
	if (sol <= sum) return;

	if (n >= N)
	{
		// 현재까지의 비용 합보다 집에 가는 비용이 더 크다면
		if (sol > sum + a[city][1])
		{
			sol = sum + a[city][1];
		}
		return;
	}

	for (i = 1; i <= N; i++)
	{
		if (chk[i]) continue;

		chk[i] = 1;
		DFS(n + 1, sum + a[city][i], i);
		chk[i] = 0;
	}
}

void main(void)
{
	scanf("%d", &N);
	int i, j;
	for (i = 1; i <= N; i++)
	{
		for (j = 1; j <= N; j++)
		{
			scanf("%d", &a[i][j]);
			if (a[i][j] == 0) a[i][j] = 0x7fffff;		
			//0인 자리에 최댓값을 준다.
		}
	}

	DFS(1, 0, 1);
	printf("%d", sol);
}
